float regulator2(float w_zad, float  wy_o)
{
 float k=1;
 float k_i=0.00001;
 float k_d=0;
 // zmienne pomocnicze
  float p,i,d,r;
float u; // uchyb regulacji
 static float u_p=0;  // uchyb regulacji w poprzenim wywo³aniu
 static float su=0; // suma minionych uchybów regulacji

u= w_zad - wy_o;   // aktualny uchyb regulacji 

// wyznaczenie sk³¹dnika proporcjonalnego (P)
p=k*u;

// wyznaczenie sk³¹dnika ca³kuj¹cego(I)
su=su+u;
i=k_i *su;

// wyznaczenie sk³¹dnika ró¿niczkuj¹cego (D)
d= k_d*(u-u_p);
u_p=u; // zapamiêtaj chwilow¹ wartoœæ b³êdów do przys³ych obliczeñ 

r= p+i+d;  // sygna³ wyjœciowy regulatora ;

if(r<-20)
	r=-20;
if(r>20)
	r=20;

return r;


}
